TimelineComponent Charts

This component is based on the timelines-chart project

  • See : https://www.npmjs.com/package/timelines-chart
  • It will generate test data if you do not supply any data.
	
  new TimeLineChart({
    targetId:"timeline"
  }).draw(function(selection){
    alert(JSON.stringify(selection));
  });
  • You can supply the group labels and tell the component to generate test data in a sequential manner 
	
    var pageGroupLabels = [
      {app:"Retail Website",bts:["Landing","Search","Checkout","Add To Cart","Payment"]},
      {app:"Order Lookup",bts:["Orders","Login","Forgot Password","Order","Lookup"]},
      {app:"Tracking",bts:["Lookup","Form","Submit","Details","Contact US"]},
      {app:"Affliate Portal",bts:["Login","Home","Account","Sales","Orders"]}
    ];

    new TimeLineChart({
        targetId:"sequential",
        options : {
            maxLineHeight:20,
            height:1200,
            width : 800,
            scaleLabel : 'Time Spent',
            groupLabels : pageGroupLabels,
            sequential:true
        }
    }).draw(function(selection){
        alert(selection);
    })
